CREATE TABLE datosUltrasonidoParcial (
    id INT AUTO_INCREMENT,
    idnodo INT,
    distancia DOUBLE,
    `presencia` VARCHAR(255),
    fechahora DATETIME,
    PRIMARY KEY (id)
);


CREATE TABLE datosMagneticoParcial (
id int AUTO_INCREMENT,
idnodo INT,
estadoPuerta VARCHAR(255),
fechahora DATETIME,
PRIMARY KEY (id));

CREATE TABLE estado (
id int AUTO_INCREMENT,
idnodo INT,
valor INT,
fechahora DATETIME,
PRIMARY KEY (id));
